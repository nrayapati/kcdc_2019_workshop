    Stash vs Unstash

    node('LINUX') {
    sh "mkdir -p output"
    writeFile file: "output/somefile", text: "Hey look, some text."
    stash name: "stash1", includes: "output/*"
    sh "rm -rf output"
    sh "ls -lrt"
}

node('KCDC') {
    sh 'ls -lrt'
    unstash name: "stash1"
    sh 'ls -lrt'
}
